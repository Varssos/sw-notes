Setup environment

Windows

  1. Install gcc/g++

    MinGW installer

  2. Add c++ extension to VS code

  3. Prepare makefile like this

CXX = g++

TARGET = thread

FLAGS = -g -Wall -std=c++17

all: $(TARGET).cpp
    $(CXX) $(FLAGS) -o $(TARGET) $(TARGET).cpp

clean:
    rm thread.exe
